# sqlite-vec:
#   Download the correct binary for each platform from:
#   https://github.com/asg017/sqlite-vec/releases
#   Place at: src-tauri/libs/vec0.{so|dylib|dll}
#   Loaded at runtime in storage.rs: conn.load_extension(...)
#
# Python (Unsloth training — desktop only):
#   pip install unsloth trl transformers datasets
#   Only needed if user wants local fine-tuning
#   Mobile falls back to CCP/Exo or disables training
#
# Tor (optional):
#   tor daemon must be running at socks5://127.0.0.1:9050
#   ViBo does not bundle Tor — user installs separately
#   Enabled via providers_tor_set(true) only if daemon is reachable
#
# Exo (optional compute scaler):
#   pip install exo
#   Started externally or via training::exo_set_endpoint()
#
# Kotlin sidecar (Koog agent):
#   Built separately, communicates via Tauri IPC invoke/listen
#   See: src/kotlin/AgentService.kt
